Quick Reference

Basic Model Creation

# Import the toolkit
import torchmodular as tm

# Define your architecture
model = tm.Sequential([
    tm.Linear(input_size, hidden_size),
    tm.BatchNorm1d(hidden_size),
    tm.ReLU(),
    tm.Dropout(0.1),
    tm.Linear(hidden_size, output_size)
])

# Setup training
optimizer = tm.Adam(model.parameters(), lr=0.001)
criterion = tm.CrossEntropyLoss()
trainer = tm.Trainer(model, optimizer, criterion)

# Train the model
trainer.fit(train_loader, val_loader, epochs=10)
